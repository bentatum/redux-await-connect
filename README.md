## Install

```
yarn add async-connect 
```

## Usage

```js
import asyncConnect from 'async-conncect'
import { compose, connect } from 'react-redux'

const enhance = compose(
  asyncConnect([{
    key: 'getUser',
    promise() {
      return client.get('user', { id: 'xyz' })
    }
  }]),
  connect(({ user }) => ({ user }))
)

export default enhance(props =>
  <div>
    {props.user.name}
  </div>
)

```
